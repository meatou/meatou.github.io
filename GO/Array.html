# 定义

语法

// 方式一
var 数组 [长度]数据类型 = [长度]数据类型{值1, 值2, 值3...}

// 方式二
var 数组 = [长度]数据类型{值1, 值2, 值3...}

// 方式三
var 数组 = [...]数据类型{值1, 值2, 值3...}

// 方式四
var 数组 = [...]数据类型{索引1:值1, 索引2:值2, 索引3:值3...}

// 方式五
var 数组 := [...]数据类型{值1, 值2, 值3...}

代码

	var array1 [3]int = [3]int{1, 2, 3}
	fmt.Println("数组1是:", array1)

	var array2 = [3]int{1, 2, 3}
	fmt.Println("数组2是:", array2)

	var array3 = [...]int{1, 2, 3}
	fmt.Println("数组3是:", array3)

	var array4 = [...]int{0:1, 1:2, 2:3}
	fmt.Println("数组4是:", array4)

	array5 := [3]int{1, 2, 3}
	fmt.Println("数组5是:", array5)

输出

数组1是: [1 2 3]
数组2是: [1 2 3]
数组3是: [1 2 3]
数组4是: [1 2 3]
数组5是: [1 2 3]

# 遍历

语法

for 索引,值 := range 数组{
    循环体
}

示例：

func main() {
	heros := [...]string{"钢铁侠", "雷神", "灭霸"}
	//遍历
	for index, value := range heros {
		fmt.Printf("索引：%v，值：%v\n", index, value)
	}
}

输出：

索引：0，值：钢铁侠
索引：1，值：雷神
索引：2，值：灭霸
