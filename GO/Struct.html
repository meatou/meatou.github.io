# 结构体定义

语法

// 声明结构体
type 结构体 struct{
    值1 值类型
    值2 值类型
    值3 值类型
    ...
}

// 方式一 
var 变量 = 结构体
// 赋值
变量.值1 = 值1
变量.值2 = 值2
...

// 方式二：
变量 := 结构体{值1,值2...}

// 方式三
var 变量 *结构体 = new(结构体)
// 赋值
(*变量).值1 = 值1
(*变量).值2 = 值2

// 方式四
var 变量 *结构体 = &结构体{值1,值2...}
...
